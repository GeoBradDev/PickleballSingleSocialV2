# SEO Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add search engine optimization and social sharing support to the React SPA: per-page meta tags, pre-rendered static HTML for public routes, JSON-LD structured data for events, and robots.txt/sitemap.xml.

**Architecture:** react-helmet-async v3 for per-page meta management, vite-prerender-plugin for build-time static HTML generation via renderToString, hardcoded route-to-meta lookup for pre-render head injection (avoids broken Helmet SSR context on React 19), EventJsonLd component for Schema.org Event structured data.

**Tech Stack:** react-helmet-async@^3.0.0, vite-prerender-plugin, react-dom/server (renderToString), react-router (StaticRouter)

**Spec:** `docs/superpowers/specs/2026-03-24-seo-design.md`

**Note:** This project has no frontend tests. TDD steps are omitted. Verification is done via build output inspection and manual checks.

**Note:** Per user policy, do NOT create git commits or push code.

### Task 6: Create SEOHead component

**Files:**
- Create: `frontend/src/components/SEOHead.jsx`

- [ ] **Step 1: Create SEOHead.jsx**

Reusable component that takes a route path and renders the appropriate Helmet tags.

```jsx
import { Helmet } from 'react-helmet-async';
import { SEO_META, SITE_URL, OG_IMAGE } from '../seoMeta.js';

function SEOHead({ path }) {
  const meta = SEO_META[path];
  if (!meta) return null;

  const url = `${SITE_URL}${path === '/' ? '' : path}`;

  return (
    <Helmet>
      <title>{meta.title}</title>
      <meta name="description" content={meta.description} />
      <link rel="canonical" href={url} />
      <meta property="og:title" content={meta.title} />
      <meta property="og:description" content={meta.description} />
      <meta property="og:url" content={url} />
      <meta property="og:image" content={OG_IMAGE} />
      <meta name="twitter:card" content="summary_large_image" />
      <meta name="twitter:title" content={meta.title} />
      <meta name="twitter:description" content={meta.description} />
      <meta name="twitter:image" content={OG_IMAGE} />
    </Helmet>
  );
}

export default SEOHead;
```

```jsx
import { Helmet } from 'react-helmet-async';

const SITE_URL = 'https://pickleballsinglessocial.com';

function EventJsonLd({ event }) {
  const startDate = new Date(event.event_date);
  const endDate = new Date(startDate.getTime() + 2.5 * 60 * 60 * 1000);
  const registered = (event.male_count || 0) + (event.female_count || 0);

    startDate: startDate.toISOString(),
    endDate: endDate.toISOString(),
    eventStatus: 'https://schema.org/EventScheduled',
    eventAttendanceMode: 'https://schema.org/OfflineEventAttendanceMode',
    location: {
      '@type': 'Place',
      name: 'Arch Pickleball',
      address: {
        '@type': 'PostalAddress',
        streetAddress: '12400 Dorsett Rd',
        addressLocality: 'Maryland Heights',
        addressRegion: 'MO',
        postalCode: '63043',
        addressCountry: 'US',
      },
    },
    description: `Singles pickleball social event for ages ${event.age_label} in St. Louis. Mixed doubles, optional coaching, and post-event matching.`,
    image: `${SITE_URL}/pickleballsinglessociallogo.png`,
    organizer: {
      '@type': 'Organization',
      name: 'Pickleball Singles Social',
      url: SITE_URL,

        : 'https://schema.org/InStock',
      url: `${SITE_URL}/events/${event.id}/register`,
    },
  };

  return (
```

- [ ] **Step 1: Create main-prerender.jsx**

```jsx
import { renderToString } from 'react-dom/server';
import { StaticRouter } from 'react-router';
import { HelmetProvider } from 'react-helmet-async';
import { ThemeProvider } from '@mui/material/styles';
import CssBaseline from '@mui/material/CssBaseline';
import theme from './theme.js';
import App from './App.jsx';
import { SEO_META, SITE_URL, OG_IMAGE } from './seoMeta.js';

const PRERENDER_ROUTES = ['/', '/about', '/faq', '/code-of-conduct', '/events'];

function buildHead(url) {
  const meta = SEO_META[url] || SEO_META['/'];
  const canonical = `${SITE_URL}${url === '/' ? '' : url}`;

  return {
    title: meta.title,
    elements: new Set([
      { type: 'meta', props: { name: 'description', content: meta.description } },
      { type: 'link', props: { rel: 'canonical', href: canonical } },
      { type: 'meta', props: { property: 'og:title', content: meta.title } },
      { type: 'meta', props: { property: 'og:description', content: meta.description } },
      { type: 'meta', props: { property: 'og:url', content: canonical } },
      { type: 'meta', props: { property: 'og:image', content: OG_IMAGE } },
      { type: 'meta', props: { name: 'twitter:card', content: 'summary_large_image' } },
      { type: 'meta', props: { name: 'twitter:title', content: meta.title } },
      { type: 'meta', props: { name: 'twitter:description', content: meta.description } },
      { type: 'meta', props: { name: 'twitter:image', content: OG_IMAGE } },
    ]),
  };
}

export async function prerender(data) {
  const url = data.url || '/';

  if (!PRERENDER_ROUTES.includes(url)) {
    return { html: '', links: [] };
  }

  const html = renderToString(
    <HelmetProvider>
      <StaticRouter location={url}>
        <ThemeProvider theme={theme}>
          <CssBaseline />
          <App />
        </ThemeProvider>
      </StaticRouter>
    </HelmetProvider>
  );

  const head = buildHead(url);

  // Extract internal links from rendered HTML for crawler
  const linkRegex = /href="(\/[^"]*?)"/g;
  const links = [];
  let match;
  while ((match = linkRegex.exec(html)) !== null) {
    const href = match[1];
    if (PRERENDER_ROUTES.includes(href) && !links.includes(href)) {
      links.push(href);
    }
  }

  return { html, head, links };
}
```

- [ ] **Step 3: Create prerender script**

Create `frontend/scripts/prerender.js`:

```javascript
import { readFileSync, writeFileSync, mkdirSync } from 'fs';
import { join, dirname } from 'path';
import { fileURLToPath } from 'url';
import { renderToString } from 'react-dom/server';
import { createElement } from 'react';

const __dirname = dirname(fileURLToPath(import.meta.url));
const DIST = join(__dirname, '..', 'dist');

const ROUTES = ['/', '/about', '/faq', '/code-of-conduct', '/events'];

async function prerender() {
  const template = readFileSync(join(DIST, 'index.html'), 'utf-8');

  // Dynamic import of the built prerender module
  const { prerender: prerenderFn } = await import(join(DIST, 'assets', 'main-prerender.js'));

  for (const route of ROUTES) {
    const result = await prerenderFn({ url: route });
    if (!result.html) continue;

    let html = template.replace('<div id="root"></div>', `<div id="root">${result.html}</div>`);

    // Inject head tags
    if (result.head?.title) {
      html = html.replace(/<title>.*?<\/title>/, `<title>${result.head.title}</title>`);
    }

    const outDir = route === '/' ? DIST : join(DIST, route.slice(1));
    mkdirSync(outDir, { recursive: true });
    writeFileSync(join(outDir, 'index.html'), html);
    console.log(`Pre-rendered: ${route}`);
  }
}
```
